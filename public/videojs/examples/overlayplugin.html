Overlay plugin


Full description of the Overlay plugin including available options you can find on Nuevodevel demo website page.

Example Code Setup


<!--Load player skin, videojs and nuevo plugin inside <body> tag -->
<link href="../skins/nuevo/videojs.min.css" rel="stylesheet" type="text/css" />
<script src="../video.min.js"></script>
<script src="../nuevo.min.js"></script>
<script src="../plugins/overlay.js"></script>

<!--Video tag setup-->
<video id="player_one" class="video-js vjs-fluid" controls preload playsinline poster="../examples/assets/images/big_buck.jpg">
	<source src="http://commondatastorage.googleapis.com/gtv-videos-bucket/sample/BigBuckBunny.mp4" type="video/mp4">
</video>

<!--Initialize the player and the plugin-->
<script>
	var player = videojs('player_one', {license: "key"});
	player.nuevo({ 
		//option_name: option_value, option_name: value
	}); 
	const overlaylist=[
		{txt:"Overlay text at top-left position.",offsetY:10, start:2,duration:8},
		{txt:"Overlay text at bottom-right position.", position: 'bottom-right',background:'#c00',start:3,duration:7},
		{txt:"Enjoy Overlay Plugin!.",color:"black",position:'top-center',offsetY:10,background:'gold',fontSize:22,start:17,duration:38},
		{img:"https://nvdata.b-cdn.net/img/cnt2.jpg",close:true,url:'https://www.nuevodevel.com/nuevo/order',start:10,position:"center",border:"white",duration:7},
		{id:"f3",video:"https://opencdn.b-cdn.net/rss/rssover.mp4",close:true,start:18,position:"center-right",offsetX:10,border:"#ccc"}		
	];
	player.overlay(overlaylist);	
</script>
