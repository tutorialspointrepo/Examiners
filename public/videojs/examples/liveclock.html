Live Clock plugin

Example Code Setup

<!--Load skin css, videojs and nuevo plugin inside <body> section-->
<link href="../skins/nuevo/videojs.min.css" rel="stylesheet" type="text/css" />
<script src="../video.min.js"></script>
<script src="../nuevo.min.js"></script>
<script src="../plugins/liveclock.min.js"></script>

<!--Video tag setup-->
<video id="player_one" class="video-js vjs-fluid" controls playsinline poster="../examples/assets/images/poster.jpg">
	<source src="path-to-live-stream.m3u8" type="application/x-mpegURL">
</video>

<!--Initialize the player and plugins-->
<script>
	var player = videojs('player_one, {license: "key"})';
	player.nuevo({ 
		//option_name: option_value, option_name: value
	}); 
	player.liveclock({seconds:true,position:"overlay"});
</script>
Check LiveClock plugin options on Nuevodevel Demo page.
