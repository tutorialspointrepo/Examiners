News Ticker Plugin


Ticker plugin full usage description, including number of options is available on Nuevodevel demo website page.

Example Code Setup


<!--Load player skin, videojs and nuevo plugin inside <body> tag -->
<link href="../skins/nuevo/videojs.min.css" rel="stylesheet" type="text/css" />
<link href="../plugins/videojs.ticker.css" rel="stylesheet" type="text/css" />
<script src="../video.min.js"></script>
<script src="../nuevo.min.js"></script>
<script src="../plugins/videojs.ticker.js"></script>

<!--Video tag setup-->
<video id="player_one" class="video-js vjs-fluid" controls preload playsinline poster="../examples/assets/images/poster.jpg">
	<source src="//opencdn.b-cdn.net/video/demo_720.mp4" type="video/mp4">
	<source src="//opencdn.b-cdn.net/video/demo_720.webm" type="video/webm">
</video>

<!--Initialize the player and the plugin-->
<script>
	const player = videojs('player_one', {license: "key"});
	player.nuevo({ 
		//option_name: option_value, option_name: value
	}); 
	player.ticker({
		rss:'../examples/assets/rss/news.xml',
		logo:'../examples/assets/images/rsslogo.png',
		logotitle:'Daily News'
	});
	
</script>
