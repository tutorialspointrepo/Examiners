Basic example setup

Basic Code Setup


<!--Load player skin, videojs and nuevo plugin inside <body> tag -->
<link href="../skins/nuevo/videojs.min.css" rel="stylesheet" type="text/css" />
<script src="../video.min.js"></script>
<script src="../nuevo.min.js"></script>

<!--Video tag setup-->
<video id="player_one" class="video-js vjs-fluid" controls preload playsinline poster="../examples/assets/images/coffee.jpg">
	<source src="../examples/assets/video/coffee.mp4" type="video/mp4">
	<source src="../examples/assets/video/coffee.webm" type="video/webm">
</video>

<!--Initialize the player and the plugin-->
<script>
	var player = videojs('player_1', {license: "key"});
	player.nuevo({ 
		//option_name: option_value, option_name: value
	}); 
</script>
IMPORTANT: It is crucial to load player JS files and initialize it from the <body> section of website, not <head>.

You can declare the nuevo plugin options in a separate array: 
<script>
    var nuevo_options = { option1: value1, option2: value2 };
	var player = videojs('player_1, {license: "key"})';
	player.nuevo(nuevo_options);
</script>
It is also possible to declare all options as player's options only. 
<!--Video tag setup-->
<video id="player_1" class="video-js vjs-fluid" controls preload playsinline></video>

<!--Initialize player-->
<script>
	const player = videojs("player_1", {
		license: "key",
		poster: "//path-to-poster-image.jpg",
		sources: {
			src: "//path-to-m3u8-playlist.m3u8",
			type: "application/x-mpegURL"
		},
		contextMemu: false,
		pipButton: false,
		buttonForward: true
    })
</script>
